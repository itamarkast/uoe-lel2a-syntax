\frame{
    \begin{itemize}
        \item We have different ways of representing \emph{structure}.
        \item The two main ones are bracket notation and trees.
        \item For example, we can start with this:
    \end{itemize}

\bigskip
    \begin{multicols}{2}
    {[}Anna] [read] [[a] [book]]

    \begin{forest}
        [{Anna read a book}
            [Anna ]
            [read ]
            [{a book}
                [a ]
                [book ]
            ]
        ]
    \end{forest}
    
    {[[}The] [students]] [love] [linguistics]

    \begin{forest}
        [{The students love linguistics}
            [{The students}
                [The ]
                [students ]
            ]
            [love ]
            [linguistics ]
        ]
    \end{forest}
    \end{multicols}
    
}
